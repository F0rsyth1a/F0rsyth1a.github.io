# 数据结构
## 并查集
### 模板 树上启发式合并
~~~cpp
struct DSU
{
	int p[N], r[N];
	void init()
	{
	    for(int i = 1; i <= n; i ++) p[i] = i, r[i] = 1;
	}
	int find(int x)
	{
	    return p[x] == x ? x : (p[x] = find(p[x]));
	}
	void merge(int a, int b) //将小树合并至大树上，树长不增加
	{
	    int x = find(a), y = find(b);
	    if(x == y) return ;
	    if(r[x] > r[y]) p[y] = x;
	    else if(r[x] < r[y]) p[x] = y;
	    else p[y] = x, r[x] ++;
	}
}
~~~
### 拓展域
构造反集  
集合{1, 2, 3}其反集{1', 2', 3'}  
若1与2敌对，merge(1, 2')，merge(1', 2)  
若1与2友好，merge(1, 2)  
通过开n*2数组，利用i + n构造反集i'  
~~~cpp
void init()
{
    for(int i = 1; i <= 2 * n; i ++) p[i] = i;
}
//按关系合并u,v
if(朋友) merge(u, v);
else //敌对
merge(u, v + n);
merge(v, u + n);
~~~
